# Biểu đồ histogram
def _plot_hist_subplot(x, fieldname, bins = 10, use_kde = True):
  x = x.dropna()
  xlabel = '{} bins tickers'.format(fieldname)
  ylabel = 'Count obs in {} each bin'.format(fieldname)
  title = 'histogram plot of {} with {} bins'.format(fieldname, bins)
  ax = sns.distplot(x, bins = bins, kde = use_kde)
  ax.set_xlabel(xlabel)
  ax.set_ylabel(ylabel)
  ax.set_title(title)
  return ax

# Biểu đồ barchart
def _plot_barchart_subplot(x, fieldname):
  xlabel = 'Group of {}'.format(fieldname)
  ylabel = 'Count obs in {} each bin'.format(fieldname)
  title = 'Barchart plot of {}'.format(fieldname)
  x = x.fillna('Missing')
  df_summary = x.value_counts(dropna = False)
  y_values = df_summary.values
  x_index = df_summary.index
  ax = sns.barplot(x = x_index, y = y_values, order = x_index)
  # Tạo vòng for lấy tọa độ đỉnh trên cùng của biểu đồ và thêm label thông qua annotate.
  labels = list(set(x))
  for label, p in zip(y_values, ax.patches):
    ax.annotate(label, (p.get_x()+0.25, p.get_height()+0.15))
  plt.xlabel(xlabel)
  plt.ylabel(ylabel)
  plt.title(title)
  return ax

MIN_VAL = -99999999
MAX_VAL = - MIN_VAL
def _bin_table(data, colname, n_bins=10, qcut=None):
    sub_data = data[[colname, 'BAD']]
    sub_data = sub_data.sort_values(colname)
    coltype = sub_data[colname].dtype
    
    if coltype in [np.float64, np.int64]:
        if qcut is not None:
            bins, thres_bin = pd.cut(sub_data[colname], bins=qcut, retbins=True)
            sub_data['bins'] = bins
        else: 
            try:
                bins, thres_bin = pd.qcut(sub_data[colname], q=n_bins, retbins=True)
                thres_bin[0] = MIN_VAL
                thres_bin[-1] = MAX_VAL
                bins, thres_bin = pd.cut(sub_data[colname], bins=thres_bin, retbins=True)
                sub_data["bins"] = bins
            except:                
                print('n_bins must be lower to bin interval is valid!')
                return None, None
    else:
        print("coltype is not float, int")
        sub_data["bins"] = sub_data[colname]
        
    df_GB = pd.pivot_table(sub_data, index=['bins'], values=['BAD'], columns=['BAD'], aggfunc={'BAD': np.size})
    
    df_summary = pd.DataFrame(index=df_GB.index)
    df_summary['#BAD'] = df_GB['BAD', 0]
    df_summary['#GOOD'] = df_GB['BAD', 1]
    df_summary['No_Obs'] = df_summary['#BAD'] + df_summary['#GOOD']
    if coltype in [np.float64, np.int64]:
        df_summary['Thres'] = thres_bin[1:]
    else: 
        thres_bin = None
        
    return df_summary, thres_bin

def _WoE(data_frame:pd.DataFrame, colname:str, n_bins:int=None, qcut:List[int]=None, min_obs:int=100):
    df_summary, thres = _bin_table(data_frame, colname, n_bins, qcut)
    df_summary["#BAD"] = df_summary["#BAD"].replace({0:1})
    
    if qcut is not None:
        exclude_index = np.where(df_summary["No_Obs"] <= min_obs)
        if len(exclude_index) > 0:
            new_thres = np.delete(thres, exclude_index)
            df_summary, thres = _bin_table(data, colname, qcut=new_thres)
            
    df_summary["%GOOD"] = df_summary["#GOOD"] / df_summary["#GOOD"].sum()
    df_summary["%BAD"] = df_summary["#BAD"] / df_summary["#BAD"].sum()
    df_summary["WOE"] = np.log(df_summary["%GOOD"] / df_summary["%BAD"])
    df_summary["IV_BIN"] = (df_summary["%GOOD"] - df_summary["%BAD"]) * df_summary["WOE"] 
    df_summary["COLUMN"] = colname
    
    IV = df_summary["IV_BIN"].sum()
    print("IV of column {} is {}".format(colname, IV))
    
    return df_summary, IV, thres

4. Dự đoán điểm của hồ sơ mới¶

Các bước thực hiện như sau:

  • Tính điểm credit score cho mỗi feature để đưa điểm xác suất về điểm tín dụng theo công thức $$\text{Score} = (\beta.\text{WOE}+\frac{\alpha}{n}).\text{Factor}+\frac{\text{Offset}}{n}$$ Trong đó:
  • $\beta$: Hệ số của biến trong phương trình hồi qui logistic.
  • $\alpha$: Hệ số chặn của phương trình hồi qui logistic.
  • $WOE$: Hệ số trọng số bằng chứng của mỗi feature.
  • $n$: Số lượng các biến của mô hình.
  • $Factor,Offset$: Là các tham số được thiết lập để tính Score.
  • Tính điểm của hồ sơ thông qua tổng điểm các feature

Tính điểm credit score cho mỗi feature

Xây dựng hàm tính creditscore cho mỗi bin của biến

In [22]:
def _CreditScore(beta, alpha, woe, n=12, odds=1/4, pdo=-50, thres_score=600):
    factor = pdo/np.log(2)
    offset = thres_score - factor*np.log(odds)
    score = (beta*woe+alpha/n)*factor+offset/n
    return score
